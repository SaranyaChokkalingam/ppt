import chromadb

def main():
    client = chromadb.Client()
    collection = client.get_collection("my_java_embeddings")

    query_text = "AI tools for Java developers"
    # Normally you'd convert query_text → embedding (using same model as in Java)
    # For demo: just search based on stored documents (text similarity is not as good)

    # You can do fulltext-like search if embedding not possible
    results = collection.query(
        query_texts=[query_text],
        n_results=3,
        include=["documents", "distances"]
    )

    print("✅ Top matching documents:")
    for doc, distance in zip(results["documents"][0], results["distances"][0]):
        print(f"- {doc} (distance: {distance})")

if __name__ == "__main__":
    main()
